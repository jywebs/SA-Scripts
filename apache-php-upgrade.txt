#Prep
mkdir /build
cd /build 
yum install epel-release -y; rpm -ihv http://vault.centos.org/6.7/os/x86_64/Packages/kernel-2.6.32-573.el6.x86_64.rpm; yum install wget gcc libmcrypt-devel openssl-devel libxml2-devel bzip2-devel curl-devel libjpeg-devel libpng-devel freetype-devel libc-client-devel.i686 libc-client-devel pcre-devel -y

# APACHE install
wget http://apache.cs.utah.edu/httpd/httpd-2.4.20.tar.gz
gunzip httpd-2.4.20.tar.gz
tar -xvf httpd-2.4.20.tar
cd httpd-2.4.20
./configure --prefix=/usr/local/apache
make && make install

#PHP Install
cd /build
wget -O php-5.6.6.tar.gz http://pl1.php.net/get/php-5.6.6.tar.gz/from/this/mirror
gzip php-5.6.6.tar.gz
tar -xvf php-5.6.6.tar
cd php-5.6.6
./configure --enable-bcmath --with-bz2 --enable-calendar --with-curl --enable-exif --enable-ftp --with-gd --with-jpeg-dir --with-png-dir --with-freetype-dir --enable-gd-native-ttf --with-imap --with-imap-ssl --with-kerberos --enable-mbstring --with-mcrypt --with-mhash --with-mysql --with-mysqli --with-openssl --with-pcre-regex --with-pdo-mysql --with-zlib-dir --with-regex --enable-sysvsem --enable-sysvshm --enable-sysvmsg --enable-soap --enable-sockets --with-xmlrpc --enable-zip --with-zlib --enable-inline-optimization --enable-mbregex --enable-opcache --enable-fpm --prefix=/usr/local/apache/php
make && make installl
